This model is designed only for logistic regression problems. You still can change the output form but you have to code it :D

class DeepNeuralNetwork:
    def __init__(self, input_shape, layers_dims, verbose=True):
        self.input_shape = input_shape
        self.layers_dims = [self.input_shape] + layers_dims + [1]
        self.params = {}
        for i in range(1, len(self.layers_dims)):
            self.params['w' + str(i)] = np.random.randn(self.layers_dims[i], self.layers_dims[i - 1]) * 0.1
            self.params['b' + str(i)] = np.zeros((self.layers_dims[i], 1))
        print('Weights initialized')
        if verbose:
            print(self.params)

    def forward(self, X):
        self.cache = {}
        self.cache['a0'] = X if len(X.shape) > 1 else np.expand_dims(X, axis=1)
        for i in range(1, len(self.layers_dims)):
            z = np.dot(self.params['w' + str(i)], self.cache['a' + str(i - 1)]) + self.params['b' + str(i)]
            self.cache['a' + str(i)] = self.sigmoid_activation(z) if i == len(self.layers_dims) - 1 else self.relu_activation(z)
        return self.cache['a' + str(len(self.layers_dims) - 1)]

    def backward(self, y_true):
        self.grads = {}
        # derivative of loss with respect to output (dA * derivative of activation function)
        m = y_true.shape[1]
        dZ = self.cache['a' + str(len(self.layers_dims) - 1)] - y_true # equivalent to dA * sigmoid_derivative(Z)
        for i in reversed(range(1, len(self.layers_dims))):
            self.grads['dw' + str(i)] = (1 / m) * np.dot(dZ, self.cache['a' + str(i - 1)].T)
            self.grads['db' + str(i)] = (1 / m) * np.sum(dZ, axis=1, keepdims=True)
            dA_prev = np.dot(self.params['w' + str(i)].T, dZ)
            dZ = dA_prev * self.relu_derivative(self.cache['a' + str(i - 1)])

    def update_params(self, learning_rate):
        for i in range(1, len(self.layers_dims)):
            self.params['w' + str(i)] -= learning_rate * self.grads['dw' + str(i)]
            self.params['b' + str(i)] -= learning_rate * self.grads['db' + str(i)]

    def train(self, X, y_true, epochs, step, learning_rate, verbose=True):
        for epoch in range(epochs):
            y_pred = self.forward(X)
            self.backward(y_true)
            self.update_params(learning_rate)
            if verbose and epoch % step == 0:
                print('Epoch {}/{}: Cost = {}'.format(epoch, epochs, self.cost(y_true, y_pred)))
                print()

    def sigmoid_activation(self, x):
        return 1 / (1 + np.exp(-x))

    def tanh_activation(self, x):
        return np.tanh(x)

    def sigmoid_derivative(self, x):
        return self.sigmoid_activation(x) * (1 - self.sigmoid_activation(x))
    
    def tanh_derivative(self, x):
        return 1 - np.tanh(x)  2

    def relu_activation(self, x):
        return np.maximum(0, x)

    def relu_derivative(self, x):
        return np.where(x > 0, 1, 0)

    def binary_cross_entropy(self, y_true, y_pred):
        return -(y_true * np.log(y_pred) + (1 - y_true) * np.log(1 - y_pred))

    def loss(self, y_true, y_pred):
        return self.binary_cross_entropy(y_true, y_pred)

    def cost(self, y_true, y_pred):
        m = y_true.shape[1]
        return (1 / m) * np.sum(self.loss(y_true, y_pred), axis=1, keepdims=True)

    def print_weights_shapes(self):
        for i in range(1, len(self.layers_dims)):
            print('w' + str(i) + ' shape : {}'.format(self.params['w' + str(i)].shape))
            print('b' + str(i) + ' shape : {}'.format(self.params['b' + str(i)].shape))

    def print_cache_shapes(self):
        for i in range(1, len(self.layers_dims)):
            print('a' + str(i) + ' shape : {}'.format(self.cache['a' + str(i)].shape))

    def accuracy(self, x, y_true):
        p = self.forward(x)
        cost = np.squeeze(self.cost(y_true, p))
        print('Train accuracy : {0: .1f}%'.format(np.mean(np.round(p) == y_true) * 100))
        print('Cost : {0: .2f}'.format(cost))
